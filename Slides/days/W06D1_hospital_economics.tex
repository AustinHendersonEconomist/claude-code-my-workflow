\begin{frame}{Hospitals as Economic Institutions}
  \begin{columns}
    \column{0.54\textwidth}
      \small
      \begin{itemize}
        \item Hospitals are the \textbf{largest single item} of health expenditure in most countries
        \item Unlike most firms, hospitals are often:
        \begin{itemize}
          \item \textbf{Not-for-profit}: what do they maximise?
          \item \textbf{Multi-product}: many services simultaneously
          \item \textbf{Physician-influenced}: clinical decisions by doctors
          \item \textbf{Insurance-reimbursed}: revenues from insurers/government
        \end{itemize}
      \end{itemize}
    \column{0.43\textwidth}
      \begin{alertblock}{Special market characteristics}
        \small
        \begin{itemize}
          \item High barriers to entry (capital, regulation)
          \item Location-specific natural monopoly
          \item Quality competition vs.\ price competition
        \end{itemize}
        \footnotesize\textbf{NZ:} Most hospitals publicly owned (Te Whatu Ora). Private
        hospitals mainly provide elective surgery.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{Hospital Objective Functions: Jacobs (1974) Taxonomy}
  \small Jacobs classifies hospital models into: \textbf{organism models} (hospital as social organism pursuing non-profit goals) and \textbf{exchange models} (coalition of interest groups bargaining over resources).
  \begin{columns}
    \column{0.48\textwidth}
      \small\textbf{Organism models:}
      \begin{enumerate}\small
        \item \textit{Quantity maximisation} (more patients subject to break-even)
        \item \textit{Quality maximisation} (best care regardless of cost)
        \item \textit{Mixed output}: quantity and quality jointly
      \end{enumerate}
    \column{0.48\textwidth}
      \small\textbf{Exchange models:}
      \begin{enumerate}\small
        \item \textit{Physician control}: doctors set the agenda
        \item \textit{Administrator control}: managers pursue budget and scale
        \item \textit{Bureaucratic model}: government direction
      \end{enumerate}
  \end{columns}
\end{frame}

\begin{frame}{Newhouse (1970): Non-Profit Hospital Model}
  \begin{columns}
    \column{0.55\textwidth}
      \small
      \begin{itemize}
        \item Hospital maximises utility over \textbf{quantity} ($Q$) and \textbf{quality} ($Z$):
              \[ \max_{Q,Z} U(Q, Z) \quad \text{s.t.\ } R(Q,Z) = C(Q,Z) \]
        \item Revenue = Costs (break-even constraint, not profit)
        \item Key prediction: hospitals tend to \emph{over-invest in quality} --- especially
              dimensions valued by \textbf{physicians and prestige} (equipment, specialist
              staff), not necessarily patients; leads to ``medical arms race''
      \end{itemize}
    \column{0.42\textwidth}
      \begin{alertblock}{The NZ implication}
        \small NZ public hospitals operate under government budget constraints rather than
        voluntary break-even. Objective function is more about \textbf{output targets} set
        by Te Whatu Ora than intrinsic quality or quantity maximisation.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{Kerr's Folly: Rewarding A While Hoping for B}
  \begin{block}{Kerr (1975): ``On the Folly of Rewarding A, While Hoping for B''}
    One of the most-cited papers in management: organisations routinely create incentive
    systems that reward the \emph{wrong} behaviours because what is measured is not what
    actually matters.
  \end{block}
  \begin{itemize}
    \item \textbf{Healthcare examples of Kerr's folly}:
    \begin{itemize}
      \item Fee-for-service $\Rightarrow$ rewards \emph{more} care (volume), but we want \emph{better} health
      \item Hospital occupancy targets $\Rightarrow$ rewards keeping beds full, not discharging patients promptly
      \item Waiting list length $\Rightarrow$ incentivises adding patients to list, not treating them
    \end{itemize}
    \item \textbf{NZ reforms context}: the 1993 purchaser--provider split was partly an attempt
          to create better incentives --- but created new Kerr-style distortions
  \end{itemize}
\end{frame}

\summaryside{
  \item Hospitals are unique: multi-product, non-profit (often), physician-influenced, insurance-reimbursed
  \item Jacobs (1974): organism models (quantity/quality maximisation) vs.\ exchange models (physician/administrator/bureaucratic control)
  \item Newhouse (1970): non-profit hospitals tend to over-invest in quality dimensions valued by prestige, not patients
  \item Kerr (1975): health care incentive systems routinely reward the wrong things --- this motivates careful system design
}
