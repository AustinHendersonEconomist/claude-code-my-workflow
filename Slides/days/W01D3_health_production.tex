\begin{frame}{The Health Production Function}
  \begin{block}{Central question for health policy}
    ``The production of health presents a central concern to the health economist and to
    public policy \ldots\ We must learn about the \emph{determinants} of health and about the
    \emph{contribution} of health care.''\\
    {\footnotesize --- Folland, Goodman \& Stano, \textit{The Economics of Health and Health Care}, 2013}
  \end{block}
  \vspace{0.6em}
  $$H \;=\; f\!\left(\,\text{HC},\; \text{nutrition},\; \text{housing},\; \text{education},\;
  \text{environment},\; \text{genetics},\; \text{lifestyle},\; \ldots\,\right)$$
  \vspace{0.2em}
  {\small $H$ = individual health stock;\enspace HC = health care inputs;\enspace many other inputs enter.}
\end{frame}

\begin{frame}{Main Determinants of Health}
  \begin{columns}
    \column{0.5\textwidth}
      \small
      \begin{enumerate}
        \item \textbf{Health care}: medicines, hospital, GP visits, surgery
        \item \textbf{Lifestyle}: smoking, diet, exercise, alcohol, sleep
        \item \textbf{Socioeconomic status}: income, education, employment
        \item \textbf{Physical environment}: air quality, water safety
        \item \textbf{Human biology}: age, sex, genetics, ethnicity
        \item \textbf{Social/community}: networks, stress, early childhood
      \end{enumerate}
    \column{0.46\textwidth}
      \begin{alertblock}{A critical insight}
        \small Health care is \textbf{only one} input. In many developed countries, additional
        HC spending produces much smaller marginal health gains than equivalent spending on
        housing, nutrition, or early childhood education.
      \end{alertblock}
      \vspace{0.2em}
      \begin{exampleblock}{NZ relevance}
        \small The \textbf{Marmot Review} and the \textbf{NZ Health Strategy} both recognise
        social determinants as primary policy targets.
      \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{Marginal Products and Health Policy}
  \begin{columns}
    \column{0.52\textwidth}
      \begin{block}{Flat-of-the-curve medicine}
        \small In \textbf{developed countries}, additional HC spending often produces very small
        additional health gains\\[0.2em]
        $\Rightarrow$ Resources may do more good if redirected to other health determinants
        (housing, nutrition, prevention)
      \end{block}
      \vspace{0.3em}
      \begin{alertblock}{Steep-of-the-curve medicine}
        \small In \textbf{developing countries}, basic HC (vaccines, clean water, obstetric care)
        still has very high marginal returns
      \end{alertblock}
    \column{0.44\textwidth}
      \small
      \begin{itemize}
        \item Health policy is concerned with \textbf{marginal products}: where does the
              \emph{next dollar} produce the most health?
        \item This is exactly what economic evaluation (Part II) aims to answer
        \item \textbf{Prevention vs.\ cure} trade-off arises here
        \item The HPF clarifies why ``more health care'' $\neq$ ``better health''
      \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{Te Tiriti o Waitangi and M\=aori Health}
  \begin{columns}
    \column{0.55\textwidth}
      \small
      \begin{itemize}
        \item M\=aori are the \textit{tangata whenua} of Aotearoa NZ; the Crown has
              obligations under \textbf{Te Tiriti o Waitangi}
        \item Substantial M\=aori health inequities: life expectancy, chronic disease
              burden, access to primary care
        \item Inequities reflect \textbf{social determinants}: historical dispossession,
              lower average incomes, housing
        \item Health economics must grapple with equity, not just efficiency
      \end{itemize}
    \column{0.42\textwidth}
      \begin{alertblock}{A recurring theme}
        \small M\=aori health equity and the \textbf{equity--efficiency trade-off}
        will recur throughout ECON306, especially in Part II.
      \end{alertblock}
      \vspace{0.3em}
      \begin{block}{Te Whatu Ora}
        \small Health New Zealand (Te Whatu Ora) formally incorporated Te Tiriti principles
        into NZ's health system in 2022.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Implications for Health Policy}
  \begin{itemize}
    \item \textbf{Health policy} is much broader than ``health care policy'' — many inputs,
          many sectors
    \item Policy tools extend beyond health:
    \begin{itemize}
      \item Housing policy (damp homes $\Rightarrow$ respiratory disease)
      \item Education policy (more education $\Rightarrow$ better health)
      \item Income support ($\Rightarrow$ better nutrition, housing, less stress)
      \item Environmental regulation (clean air $\Rightarrow$ fewer respiratory conditions)
    \end{itemize}
    \item \textbf{Week 3}: given market failures and the complex HPF, what is the
          appropriate scope of government intervention?
  \end{itemize}
  \begin{exampleblock}{NZ example: Healthy Homes Standards (2019)}
    Minimum insulation, heating, and ventilation in rental properties --- housing quality
    recognised as a health input; market failures justify regulation.
  \end{exampleblock}
\end{frame}

\summaryside{
  \item Health is produced by many inputs, not just health care: lifestyle, housing, education, environment, biology
  \item In developed countries, we are often on the ``flat of the curve'' --- additional HC spending has low marginal health returns
  \item Health \emph{policy} is about comparing marginal products and directing resources where they do the most good
  \item Equity dimensions --- including M\=aori health in NZ --- are central to any complete analysis of health economics
}
