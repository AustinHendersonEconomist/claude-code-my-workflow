\begin{frame}{Classifying Health Systems}
  \begin{columns}
    \column{0.50\textwidth}
      \small\textbf{Key design dimensions:}
      \begin{enumerate}\small
        \item \textbf{Financing}: taxes, payroll contributions, premiums, out-of-pocket
        \item \textbf{Purchasing}: government, social insurance fund, private insurers
        \item \textbf{Provision}: public, private (not-for-profit, for-profit)
        \item \textbf{Universality}: universal, employment-based, residual
      \end{enumerate}
    \column{0.47\textwidth}
      \small\textbf{Three ideal types (with many hybrids):}
      \begin{itemize}\footnotesize
        \item \textbf{Private (US)}: voluntary; employment-based; fragmented; residual safety net
        \item \textbf{Social insurance (Bismarck)}: payroll-funded statutory funds; universal via mandates; Germany, France, Japan
        \item \textbf{Public (Beveridge)}: tax-funded; universal; government provider; UK NHS, NZ, Canada
      \end{itemize}
  \end{columns}
\end{frame}

\begin{frame}{The NZ Health System: Structure}
  \begin{columns}
    \column{0.54\textwidth}
      \footnotesize
      \begin{itemize}
        \item \textbf{Beveridge-style} universal public system; funded by general taxation (Vote: Health)
        \item \textbf{Health NZ $|$ Te Whatu Ora} (est.\ 2022): single national authority replacing 20 DHBs
        \item Primary care: GPs enrolled with PHOs under capitation funding
        \item Pharmaceuticals: \textbf{PHARMAC} cost-effectiveness funding
        \item ACC: universal no-fault accident insurance (no tort system)
        \item Private sector: Southern Cross + other insurers for electives
      \end{itemize}
    \column{0.43\textwidth}
      \begin{alertblock}{Ageing and the NZ system}
        \small The \textit{localities} framework responds to the growing need for \textbf{integrated aged care} as the population ages.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{NZ Health System: History of Reforms (1938--2022)}
  \begin{columns}
    \column{0.32\textwidth}
      \begin{block}{1938--1983}
        \footnotesize\textbf{Welfare state foundations:} 1938 Social Security Act — universal, tax-funded health care; govt-employed doctors; stable but cost-pressured.
      \end{block}
    \column{0.32\textwidth}
      \begin{block}{1984--1999}
        \footnotesize\textbf{Market reforms:} 1993 purchaser--provider split; CHEs (semi-commercial hospitals) + RHAs (purchasing agents); intended to introduce competition; largely failed.
      \end{block}
    \column{0.32\textwidth}
      \begin{alertblock}{2000--2022}
        \footnotesize\textbf{Re-integration:} 2001 DHBs replace RHAs/CHEs; 2022 Te Whatu Ora replaces all 20 DHBs; rationale: equity, reduce duplication, integrate aged care.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{Why Did the Purchaser--Provider Split Fail?}
  \begin{columns}
    \column{0.54\textwidth}
      \small
      \begin{itemize}
        \item \textbf{Contract failure}: healthcare too complex to fully specify in contracts
        \item \textbf{Transaction costs}: vast resources diverted to contracting and monitoring
        \item \textbf{Hospital behaviour}: public hospitals didn't act like profit-maximising firms
        \item \textbf{Equity}: market mechanisms produced geographic/socioeconomic inequalities
        \item \textbf{Political economy}: voters and professionals rejected commercialisation
      \end{itemize}
    \column{0.43\textwidth}
      \begin{alertblock}{Ashton, Mays \& Devlin (2005)}
        \small ``Continuity through change'' --- the rhetoric of reform exceeded the reality.
      \end{alertblock}
      \begin{exampleblock}{Tutorial: Logan (1986)}
        \small ``Bob's Party'' --- satire presciently relevant to NZ's 1993 reforms.
      \end{exampleblock}
  \end{columns}
\end{frame}

\summaryside{
  \item Three health system types: private insurance (US), social insurance (Bismarck), public (Beveridge/NZ)
  \item NZ: Beveridge-style, tax-funded, universal, now administered by Health New Zealand (Te Whatu Ora)
  \item NZ history: 1938 welfare state $\to$ 1993 purchaser--provider split (failed) $\to$ 2022 re-centralisation
  \item Kerr's folly applies: reform-era incentive systems rewarded the wrong things
}
