\recapslide{
  \item Expected utility theory: risk-averse individuals prefer a certain outcome over a fair gamble
  \item Moral hazard (ex-ante and ex-post): insurance changes incentives
  \item Adverse selection: high-risk individuals disproportionately seek insurance $\Rightarrow$ potential market failure
}

\planslide{Public vs.\ Private Insurance; International Insurance Systems}{
  \item Comparison of public and private insurance design
  \item NZ's hybrid insurance model: public + Southern Cross + ACC
  \item Community rating, mandatory coverage, and risk equalisation
}

\begin{frame}{Public vs.\ Private Insurance Design}
  \begin{columns}
    \column{0.48\textwidth}
      \begin{block}{Public insurance}
        \small
        \begin{itemize}
          \item Financed by taxes; mandatory for all
          \item Resolves adverse selection via community rating
          \item More egalitarian; lower admin costs; monopsony power
        \end{itemize}
      \end{block}
    \column{0.48\textwidth}
      \begin{alertblock}{Private insurance}
        \small
        \begin{itemize}
          \item Financed by premiums; voluntary
          \item Subject to adverse selection unless regulated
          \item More choice; higher admin costs, fragmented risk pools
        \end{itemize}
      \end{alertblock}
  \end{columns}
  \begin{exampleblock}{NZ's hybrid model}
    \small NZ provides \textbf{universal public cover} for most services (via Te Whatu Ora),
    with \textbf{Southern Cross} ($\approx$1 million members) providing faster access to
    elective procedures.
  \end{exampleblock}
\end{frame}

\begin{frame}{Southern Cross and Private Insurance in NZ}
  \begin{columns}
    \column{0.54\textwidth}
      \small
      \begin{itemize}
        \item \textbf{Southern Cross}: NZ's largest private health insurer ($\approx$1 million
              members), structured as a \textbf{not-for-profit mutual society}
        \item Covers elective procedures and specialist consultations --- not competing with
              public cover for emergency/acute care
        \item Employer premium subsidies common --- tax treatment matters for uptake
        \item \textbf{Economic question}: Does private insurance \emph{complement} or
              \emph{substitute} for public provision in NZ?
      \end{itemize}
    \column{0.43\textwidth}
      \begin{alertblock}{Ageing and private insurance}
        \small As membership ages, adverse selection and rising costs create ``death spiral''
        dynamics in microcosm. Sustainable premium design requires careful risk management.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{ACC: NZ's Unique Accident Insurance}
  \begin{columns}
    \column{0.54\textwidth}
      \footnotesize
      \begin{itemize}
        \item \textbf{ACC}: universal, no-fault accident insurance for all NZ residents ---
              covers treatment costs and lost income regardless of fault
        \item Funded by levies on employers, employees, and vehicle registration
        \item Eliminates tort system; mandatory (resolves adverse selection); single
              insurer (no admin fragmentation)
        \item \textbf{Efficiency concern}: does no-fault insurance create moral hazard?
              ACC workplace safety programmes address this.
      \end{itemize}
    \column{0.43\textwidth}
      \begin{block}{Market failure framework}
        \small Natural monopoly + adverse selection + moral hazard $\Rightarrow$ public
        universal single-payer insurance is the efficient response.
      \end{block}
  \end{columns}
\end{frame}

\summaryside{
  \item Public insurance solves adverse selection via mandates and community rating, but faces political economy constraints
  \item Private insurance offers more choice but is prone to adverse selection and market fragmentation
  \item NZ's hybrid: public cover for all + private (Southern Cross) for elective/specialist speed and choice
  \item ACC is NZ's universal no-fault accident insurer --- a world-leading response to natural monopoly and adverse selection
}
