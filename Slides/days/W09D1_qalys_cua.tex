\begin{frame}{What is a QALY?}
  \small
  A \textbf{Quality-Adjusted Life Year (QALY)} = years lived $\times$ quality weight $u \in [0,1]$:
  \[ \text{QALYs} = \int_0^T u(t)\, dt \;\approx\; \sum_t u_t \cdot \Delta t \]
  \vspace{0.1em}
  \begin{columns}
    \column{0.55\textwidth}
      \begin{itemize}
        \item Perfect health $= 1.0$ QALY/year
        \item Moderate disability ($u = 0.6$) $= 0.6$ QALY/year
        \item Death $= 0$ QALY (anchor point)
        \item States $u < 0$ = \textbf{worse than death} (severe pain, complete dependency)
      \end{itemize}
    \column{0.42\textwidth}
      \begin{alertblock}{Why QALYs?}
        QALYs allow comparison across different diseases on a common scale --- compare a hip replacement and a blood pressure drug directly.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{The EQ-5D-5L Instrument}
  \begin{columns}
    \column{0.55\textwidth}
      \begin{block}{Five dimensions, five levels each}
        \begin{enumerate}
          \item \textbf{Mobility}: no problems $\to$ unable to walk
          \item \textbf{Self-care}: no problems $\to$ unable to wash/dress
          \item \textbf{Usual activities}: no problems $\to$ unable to perform
          \item \textbf{Pain/discomfort}: none $\to$ extreme
          \item \textbf{Anxiety/depression}: none $\to$ extreme
        \end{enumerate}
      \end{block}
      In total: $5^5 = 3{,}125$ possible health states
    \column{0.42\textwidth}
      \begin{alertblock}{The NZ EQ-5D value set}
        The \textbf{NZ EQ-5D-5L value set} was generated by Otago Economics-affiliated researchers
        (Devlin, Hansen et al.) using Time Trade-Off (TTO) methods with NZ population samples.\\[0.3em]
        PHARMAC uses the NZ value set for funded medicines evaluations.
      \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}{Methods for Eliciting Health State Utility Weights}
  \begin{columns}
    \column{0.32\textwidth}
      \begin{block}{Visual Analogue Scale (VAS)}
        \small Rate health state on a 0--100 ``feeling thermometer''\\[0.2em]
        \textit{Pros}: easy, fast\\[0.1em]
        \textit{Cons}: no explicit trade-off; may not reflect preferences
      \end{block}
    \column{0.32\textwidth}
      \begin{alertblock}{Time Trade-Off (TTO)}
        \small How many years in perfect health do you value equally to $T$ years in health state $X$?\\[0.2em]
        \textit{Pros}: intuitive; anchored at death\\[0.1em]
        \textit{Cons}: time preference confounds
      \end{alertblock}
    \column{0.32\textwidth}
      \begin{block}{Standard Gamble (SG)}
        \small Probability $p$ of perfect health vs.\ $(1-p)$ death equivalent to health state $X$?\\[0.2em]
        \textit{Pros}: grounded in EU theory\\[0.1em]
        \textit{Cons}: difficult; risk attitude confounds
      \end{block}
  \end{columns}
  \vspace{0.3em}
  {\footnotesize \textbf{International standard}: EuroQol Group recommends \textbf{TTO} for value set generation. The NZ EQ-5D-5L value set uses hybrid TTO.}
\end{frame}

\begin{frame}{Applying CUA: The Incremental Cost-Effectiveness Ratio (ICER)}
  \begin{columns}
    \column{0.55\textwidth}
      \small
      $\text{ICER} = \dfrac{\Delta \text{Cost}}{\Delta \text{QALYs}} = \dfrac{C_{\text{new}} - C_{\text{comp}}}{Q_{\text{new}} - Q_{\text{comp}}}$
      \vspace{0.2em}
      \begin{itemize}
        \item Compare new intervention to best \textbf{comparator} (standard of care)
        \item ICER $<$ threshold $\Rightarrow$ fund; $>$ threshold $\Rightarrow$ do not fund
        \item \textbf{NZ PHARMAC}: $\approx$\$45--50K/QALY (implicit)
        \item \textbf{UK NICE}: £20--30K;\quad \textbf{ICER (US)}: \$100--150K/QALY
      \end{itemize}
    \column{0.42\textwidth}
      \begin{exampleblock}{Why do thresholds differ?}
        \small The threshold = \textbf{opportunity cost} of the marginal health dollar. Richer countries have higher thresholds; NZ's lower threshold reflects GDP per capita.
      \end{exampleblock}
  \end{columns}
\end{frame}

\begin{frame}{Worked ICER Example}
  \begin{block}{Drug A vs.\ standard care for condition X}
    \small
    \begin{center}
      \begin{tabular}{@{}lcc@{}}
        \toprule
        & \textbf{Drug A} & \textbf{Standard care} \\
        \midrule
        Annual cost (per patient) & \$18,000 & \$8,000 \\
        QALYs gained (per patient, 5 years) & 3.2 & 2.6 \\
        \bottomrule
      \end{tabular}
    \end{center}
    \vspace{0.2em}
    Total cost difference over 5 years: $5 \times \$10{,}000 = \$50{,}000$. QALY gain: $3.2 - 2.6 = 0.6$.
    \[ \text{ICER} = \frac{\$50{,}000}{0.6} \approx \$83{,}333 \text{ per QALY} \]
    ICER $> \$50{,}000$ PHARMAC threshold $\Rightarrow$ \textbf{do not fund} under current threshold.
  \end{block}
\end{frame}
