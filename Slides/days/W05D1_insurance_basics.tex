\begin{frame}{Why Does Health Insurance Exist?}
  \begin{block}{The fundamental problem: uncertainty}
    Individuals do not know when they will become ill, or how much it will cost to treat them.
    Without insurance, they face potentially \textbf{catastrophic} and \textbf{unaffordable}
    out-of-pocket costs.
  \end{block}
  \begin{itemize}
    \item \textbf{Risk pooling}: by pooling the risks of many individuals, insurance converts
          an unpredictable large loss into a predictable small premium
    \item \textbf{Actuarially fair premium}: $\pi = p \cdot L$, where $p$ = probability of
          illness and $L$ = cost of treatment
    \item Risk-averse individuals are willing to pay \emph{more} than the actuarially fair
          premium to avoid the risk of a large loss --- the \textbf{risk premium}
    \item This creates a mutually beneficial exchange: insured pays premium $>$ expected loss;
          insurer bears the risk
  \end{itemize}
\end{frame}

\begin{frame}{Risk Aversion and Expected Utility}
  \begin{columns}
    \column{0.55\textwidth}
      \begin{block}{Expected Utility (von Neumann--Morgenstern)}
        \small With \textbf{concave utility} $U(W)$, a certain outcome is preferred to a fair
        gamble: $U(\bar{W}) > p \cdot U(W_H) + (1-p) \cdot U(W_L)$, where $W_H$, $W_L$ are
        wealth levels with/without illness.
      \end{block}
    \column{0.42\textwidth}
      \small
      \begin{itemize}
        \item Concavity degree = risk aversion degree
        \item \textbf{Risk neutral}: linear $U(W)$ $\Rightarrow$ indifferent
        \item \textbf{Risk loving}: convex $U(W)$ $\Rightarrow$ prefers gamble
        \item \textbf{Arrow-Pratt}: $-U''(W)/U'(W)$
      \end{itemize}
  \end{columns}
  \begin{alertblock}{Ageing \& risk aversion}
    \small Older individuals typically exhibit \textbf{greater} risk aversion for health outcomes
    --- consistent with rising insurance demand with age.
  \end{alertblock}
\end{frame}

\begin{frame}{Moral Hazard: Two Types}
  \begin{columns}
    \column{0.50\textwidth}
      \begin{alertblock}{Ex-ante moral hazard}
        \small Less effort to prevent illness (exercise less, take more risks). Hard to observe;
        addressed by exclusions for self-inflicted harm and wellness incentives.
      \end{alertblock}
    \column{0.48\textwidth}
      \begin{alertblock}{Ex-post moral hazard}
        \small Once insured, consume more HC (effective price $<$ full cost). Pauly (1968): a
        \emph{normal} price response. Addressed by copayments, deductibles, gatekeeping.
      \end{alertblock}
  \end{columns}
  \begin{exampleblock}{NZ example: GP co-payments}
    \small NZ primary care visits involve a co-payment ($\approx$\$20--50) even for Community
    Services Card holders --- a deliberate moral hazard reduction mechanism.
  \end{exampleblock}
\end{frame}

\begin{frame}{Adverse Selection and Insurance Market Failure}
  \begin{columns}
    \column{0.54\textwidth}
      \small
      \begin{itemize}
        \item \textbf{Adverse selection}: high-risk individuals more likely to seek insurance
        \item Insurer cannot distinguish risk types $\Rightarrow$ charges \textbf{pooled premium}
        \item Low-risk individuals find this too expensive $\Rightarrow$ opt out
        \item Premium rises further $\Rightarrow$ ``death spiral''
        \item Extreme: only highest-risk remain $\Rightarrow$ market unravels
      \end{itemize}
    \column{0.43\textwidth}
      \begin{alertblock}{Rothschild--Stiglitz (1976)}
        \small In competitive markets, equilibrium may involve:
        \begin{itemize}
          \item \textbf{Separating equilibria}: contracts screen risk types
          \item \textbf{No equilibrium}: market destroyed
        \end{itemize}
      \end{alertblock}
      \small\textbf{Policy response:} community rating (uniform premiums), mandatory coverage,
      public provision
  \end{columns}
\end{frame}

\summaryside{
  \item Risk-averse individuals pay a risk premium above the actuarially fair rate for insurance
  \item Ex-ante and ex-post moral hazard both arise from insurance; addressed by copayments, deductibles, gatekeeping
  \item Adverse selection can cause private insurance markets to unravel --- a key rationale for public health insurance
  \item NZ's primary care co-payment structure is a deliberate design choice balancing access with moral hazard reduction
}
