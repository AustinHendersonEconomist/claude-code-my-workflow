\begin{frame}{Cross-Cutting Analytical Tools}
  \begin{enumerate}
    \item \textbf{Positive/normative distinction}: used in every topic. Always ask: ``Is this
          a factual claim or a value judgement?''
    \vspace{0.3em}
    \item \textbf{Cost-benefit / cost-effectiveness reasoning}: from market failure analysis
          (Part I) through CUA/VSL (Part II) to NPV/IRR of education (Part III)
    \vspace{0.3em}
    \item \textbf{Human capital framework}: first the health production function $H = f(\text{HC},\ldots)$;
          then human capital model of education (Becker/Mincer); same investment logic
    \vspace{0.3em}
    \item \textbf{Asymmetric information and principal-agent models}: in health (SID, insurance),
          in evaluation (proxy QALYs), and in education (signalling)
    \vspace{0.3em}
    \item \textbf{NZ institutional context}: PHARMAC, ACC, Te Whatu Ora, Student Loan Scheme,
          EQ-5D NZ value set, 1000Minds --- all are examinable
  \end{enumerate}
\end{frame}
